angular.module('demo').filter('formJson', function(){
  var internalKeys;
  internalKeys = {
    '$parsers': true,
    '$formatters': true,
    '$viewChangeListeners': true
  };
  function copyWithoutNgModelInternal(it){
    return Object.keys(it).filter(function(it){
      return !internalKeys[it];
    }).reduce(function(copy, key){
      var value;
      value = it[key];
      copy[key] = angular.isObject(value) ? copyWithoutNgModelInternal(value) : value;
      return copy;
    }, {});
  }
  return function(it){
    return JSON.stringify(copyWithoutNgModelInternal(it), null, 4);
  };
}).controller('MatchCtrl', (function(){
  var prototype = constructor.prototype;
  constructor.$inject = ['$scope'];
  function constructor($scope){
    this.$scope = $scope;
  }
  return constructor;
}()));
